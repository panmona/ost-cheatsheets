\newcommand{\sqljoin}[3]{
    \begin{multicols*}{2}
        \begin{venndiagram2sets}
            #1
        \end{venndiagram2sets}
        \vfill\columnbreak
        \hspace{-8em}
        \begin{tabular}{ll}
            x & y \\
            \hline
            #2
        \end{tabular}
    \end{multicols*}
    \mint{sql}{#3}
}
